508void MainWindow::activityLevel(QCPRange range)
509{
510 double x_axis_range_lower = range.lower;
511 double x_axis_range_upper = range.upper;
512 int count = 0;
513 double sum = 0.0;
514
515 QCPGraphDataContainer::const_iterator begin =
516 ui->plot->graph(0)->data()->constBegin();
517 QCPGraphDataContainer::const_iterator end =
518 ui->plot->graph(0)->data()->constEnd();
519
520 while (begin != end)
521 {
522 if (begin->key > x_axis_range_upper)
523 break;
524 if (begin->key > x_axis_range_lower)
525 {
526 sum += begin->value;
527 count++;
528 }
529 ++begin;
530 }
531 if (count)
532 {
533 ui->Activitylevel->setText(QString::number((sum / count), 'f', 2));
534 }
535}

728 const QString format = "MM/dd/yyyy hh:mm:ss";
729
730 // dump activity/pressure data
731
732 QCPGraphDataContainer::const_iterator activity_begin = ui->plot->graph(0)->data()->findBegin(x_axis_range_lower);
733 QCPGraphDataContainer::const_iterator activity_end = ui->plot->graph(0)->data()->findEnd(x_axis_range_upper);
734 QCPGraphDataContainer::const_iterator temperature_begin = ui->plot->graph(1)->data()->findBegin(x_axis_range_lower);
735 QCPGraphDataContainer::const_iterator temperature_end = ui->plot->graph(1)->data()->findEnd(x_axis_range_upper);
736 QCPGraphDataContainer::const_iterator voltage_begin = ui->plot->graph(2)->data()->findBegin(x_axis_range_lower);
737 QCPGraphDataContainer::const_iterator voltage_end = ui->plot->graph(2)->data()->findEnd(x_axis_range_upper);
738
739 int32_t last_min = INT32_MAX;
740
741 while ((activity_begin != activity_end) ||
742 (temperature_begin != temperature_end) ||
743 (voltage_begin != voltage_end))
744 {
745 int32_t min = INT32_MAX;
746 int32_t activity_key = (activity_begin != activity_end)? activity_begin->key : INT32_MAX;
747 int32_t temperature_key = (temperature_begin != temperature_end)? temperature_begin->key : INT32_MAX;
748 int32_t voltage_key = (voltage_begin != voltage_end)? voltage_begin->key : INT32_MAX;
749 min = activity_key < min ? activity_key : min;
750 min = temperature_key < min ? temperature_key : min;
751 min = voltage_key < min ? voltage_key : min;
752
753 if (min == INT32_MAX)
754 break;
755
756 if (activity_key == last_min) {
757 ui->te_fileinfo->appendPlainText("Error: Repeated activity data time!");
758 }
759
760 last_min = min;
761
762 out << min << "," << QDateTime::fromSecsSinceEpoch(qint64(min), timezone).toString(format) << ",";
763
764 if (min == activity_key) {
765 out << QString::number(activity_begin->value, 'f', 2);
766 ++activity_begin;
767 }
768 out << ",";
769 if (min == temperature_key) {
770 out << QString::number(temperature_begin->value, 'f', 2);
771 ++temperature_begin;
772 }
773 out << ",";
774 if (min == voltage_key) {
775 out << QString::number(voltage_begin->value, 'f', 2);
776 ++voltage_begin;
777 }
778 out << "\n";
779
780 }
781 file.close();
782}

939 qint64 stop_time = stop.toSecsSinceEpoch();
940 int count = 0;
